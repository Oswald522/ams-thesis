\chapter*{符号表}
\begin{description}[itemindent=3cm,labelwidth=3cm]
  \item[$\bm{r}$]  垂直光轴的平面向量
  \item[$\bm{u}$]  控制电压向量
  \item[$\phi(\bm{r}),\phi(x,y)$]  波前畸变相位
  \item[$m(\bm{r}),m(x,y)$] 波前校正器引入的相位
  \item[$\mu(\bm{r}),\mu(x,y)$] 校正后残余波面
  \item[$D$] 望远镜直径
  \item[$C_n^2$] 折射率结构常数
  \item[$r_0$] 大气相干长度
  \item[$l_0$] 湍流内尺度
  \item[$L_0$] 湍流外尺度
  \item[$J$] 性能评价函数
  \item[$\gamma$] 增益系数
  \item[$\delta$] 扰动幅度
  \item[$\lambda$] 光波波长
\end{description}
